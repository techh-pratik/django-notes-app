pipeline{
    agent {label "vinod"}
    
    stages{
        
        stage("In this we are clonning the code"){
            steps{
                echo "This is clonning the code"
                git url: "https://github.com/techh-pratik/django-notes-app.git", branch: "main"
            }
        }
        stage("In this we are building the code"){
            steps{
                echo "This is Building the code"
                sh "docker build -t notes-app ."
            }
        }
        stage("In this we are Pushing the code"){
            steps{
                echo "This is Pushing the code"
                withCredentials([usernamePassword(
                    credentialsId:"DockerHubCred",
                    usernameVariable:"dockerHubUser", 
                    passwordVariable:"dockerHubPass")]){
                sh "docker login -u ${env.dockerHubUser} -p ${env.dockerHubPass}"
                sh "docker image tag notes-app ${env.dockerHubUser}/notes-app"
                sh "docker ${env.dockerHubUser}/push notes-app"
            }
        }
        stage("In this we are Deploying the code"){
            steps{
                echo "This is Deploying the code"
                sh "docker compose up -d"
            }
        }
    }
}
